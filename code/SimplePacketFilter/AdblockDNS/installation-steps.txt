# Install bind. 
# For Fedora 
yum install bind bind-utils
# For ubuntu
apt-get install bind9

# Now copy the config files in the appropriate location
export NAMED_PATH=<PATH_TO_NAMED_FILES>
# For ubuntu/debian based systems /etc/bind/
# For fedora it is /etc/

# Add the following line to the named.conf in ${NAMED_PATH}
include "<PATH_TO_NAMED_FILES>/named.conf.adblock";
# Copy the null.zone.file and named.conf.adblock to ${NAMED_PATH}
cp null.zone.file named.conf.adblock ${NAMED_PATH}
